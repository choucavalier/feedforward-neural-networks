\begin{frame}{Forward propagation}
\SetKwProg{Fn}{}{}{end}\SetKwFunction{ForwardProp}{ForwardPropagation}
\begin{algorithm}[H]
\Fn(){\ForwardProp{x, w}}{
$o^1 \gets x$\;
\For{l = 2 to L}{
$x = add\_bias(o^{l-1})$\;
$o^l = g((w^l)^T x)$\;
}
$return~o$\;
}
\end{algorithm}
\end{frame}

\begin{frame}
  \begin{center}
    \begin{tikzpicture}[shorten >=1pt,->,draw=black!50, node distance=2.5cm]
      \tikzstyle{every pin edge}=[<-,shorten <=1pt]
      \tikzstyle{neuron}=[circle,draw,minimum size=17pt,inner sep=0pt]

      % input layer nodes
      \foreach \y in {1,...,4}
      \node[neuron, pin=left:$x\y$] (I-\y) at (0,-\y) {};

      % hidden layer nodes
      \foreach \y in {1,...,5}
      \path[yshift=0.5cm]
      node[neuron] (H-\y) at (2.5,-\y) {};

      % output layer node
      \foreach \y in {1,2}
      \path[yshift=-1cm]
      node[neuron,pin={[pin edge={->}]right:$o\y$}] (O-\y) at (4.5,-\y) {};

      % Connect every node in the input layer with every node in the
      % hidden layer.
      \foreach \src in {1,...,4}
      \foreach \dst in {1,...,5}
      \path[draw=black,thick] (I-\src) edge (H-\dst);

      % Connect every node in the hidden layer with the output layer
      \foreach \src in {1,...,5}
      \foreach \dst in {1,2}
      \path (H-\src) edge (O-\dst);
    \end{tikzpicture}
  \end{center}
\end{frame}
